## 6. Providers Riverpod (RecommandÃ©)

// Organisation des providers par feature

// lib/features/stores/presentation/providers/stores_providers.dart

/// ============================================================
/// STORES PROVIDERS - Architecture Riverpod
/// ============================================================

// 1. DataSource Provider
final storesRemoteDataSourceProvider = Provider<StoresRemoteDataSource>((ref) {
  return StoresRemoteDataSourceImpl();
});

// 2. Repository Provider
final storesRepositoryProvider = Provider<StoresRepository>((ref) {
  return StoresRepositoryImpl(
    remoteDataSource: ref.read(storesRemoteDataSourceProvider),
  );
});

// 3. Use Case Providers
final getStoresUseCaseProvider = Provider<GetStoresUseCase>((ref) {
  return GetStoresUseCase(ref.read(storesRepositoryProvider));
});

final searchStoresUseCaseProvider = Provider<SearchStoresUseCase>((ref) {
  return SearchStoresUseCase(ref.read(storesRepositoryProvider));
});

// 4. State Provider (pour l'UI)
final storesStateProvider = StateNotifierProvider<StoresNotifier, StoresState>((ref) {
  return StoresNotifier(ref.read(getStoresUseCaseProvider));
});

// 5. Computed/Derived Providers
final storeByIdProvider = Provider.family<StoreEntity?, String>((ref, id) {
  final state = ref.watch(storesStateProvider);
  return state.stores.firstWhereOrNull((s) => s.id == id);
});
